\subsection{A2 Temporal-Spectral-Spatial Transformer (核心结构)}

\begin{lstlisting}[language=Python,caption={A2 TemporalSpectralSpatialTransformer 三支路融合}]
class TemporalSpectralSpatialTransformer(nn.Module):
    """Temporal, spectral, and channel-spatial branches with late transformer fusion."""
    def __init__(self, channels=64, timesteps=250, output_dim=512,
                 hidden_dim=256, dropout=0.2):
        super().__init__()
        # Temporal branch: multiscale ConvTransformer
        self.temporal = ConvTransformerEncoder(
            channels=channels, timesteps=timesteps,
            output_dim=output_dim, hidden_dim=hidden_dim,
            layers=4, heads=8, multiscale=True, dropout=dropout)
        # Spectral branch: STFT + CNN
        self.spectral = SpectrogramCNNEncoder(
            channels=channels, timesteps=timesteps,
            output_dim=output_dim, hidden_dim=hidden_dim // 2,
            dropout=dropout)
        # Spatial/Channel branch: learned attention over channels
        self.spatial = nn.Sequential(
            nn.Linear(timesteps, hidden_dim), nn.GELU(),
            nn.LayerNorm(hidden_dim))
        self.spatial_pool = AttentionPool1d(hidden_dim)
        # Late fusion transformer
        fusion_layer = nn.TransformerEncoderLayer(
            d_model=output_dim, nhead=8,
            dim_feedforward=output_dim * 2,
            dropout=dropout, activation="gelu",
            batch_first=True, norm_first=True)
        self.fusion_transformer = nn.TransformerEncoder(
            fusion_layer, num_layers=2)
        self.branch_type = nn.Parameter(
            torch.zeros(1, 3, output_dim))
        self.projector = nn.Sequential(
            nn.LayerNorm(output_dim),
            nn.Linear(output_dim, output_dim), nn.GELU(),
            nn.Linear(output_dim, output_dim),
            nn.LayerNorm(output_dim))

    def forward(self, eeg):
        temporal = self.temporal(eeg)      # [B, 512]
        spectral = self.spectral(eeg)      # [B, 512]
        spatial = self.spatial_out(
            self.spatial_pool(self.spatial(eeg)))  # [B, 512]
        tokens = torch.stack(
            [temporal, spectral, spatial], dim=1)
        tokens = tokens + self.branch_type
        tokens = self.fusion_transformer(tokens)
        return self.projector(
            AttentionPool1d(output_dim)(tokens))
\end{lstlisting}

\subsection{Gated Fusion 模块}

\begin{lstlisting}[language=Python,caption={GatedResidual Fusion}]
class GatedFusion(nn.Module):
    def __init__(self, dim=512, hidden_dim=None):
        super().__init__()
        hidden = hidden_dim or dim * 2
        self.gate = nn.Sequential(
            nn.Linear(dim * 2, hidden), nn.GELU(),
            nn.Linear(hidden, dim), nn.Sigmoid())
        self.eeg_delta = nn.Sequential(
            nn.LayerNorm(dim), nn.Linear(dim, hidden),
            nn.GELU(), nn.Linear(hidden, dim))

    def forward(self, image_emb, eeg_emb=None):
        if eeg_emb is None:
            return image_emb
        gate = self.gate(torch.cat([image_emb, eeg_emb], dim=-1))
        delta = self.eeg_delta(eeg_emb)
        return image_emb + gate * delta
\end{lstlisting}

    gate_proj(cat([pool(visual_tokens),
                   pool(eeg_tokens)])))

    query=visual_tokens,
    key=eeg_tokens,
    value=eeg_tokens)
enhanced_visual_tokens = (
    visual_tokens + beta * eeg_context)
